STRING — Python Syntax Card

Access & Slice

s[0]      # first char
s[-1]     # last char
s[1:4]    # index 1,2,3
s[::-1]   # reversed copy
len(s)    # length O(1)

Immutable — Modify via list

# s[0] = 'H'  -> TypeError!
a = list(s)
a[0] = 'H'
s = ''.join(a)   # back to string

Search

"lo" in s             # True/False   O(n)
s.find("ll")          # index or -1
s.index("ll")         # index or ValueError
s.count("l")          # occurrences
s.startswith("hel")   # True/False
s.endswith("lo")      # True/False

Case & Strip

s.upper(); s.lower()
s.strip()    # remove whitespace both ends
s.lstrip(); s.rstrip()
s.strip("x") # strip specific char

Split & Join

"a,b,c".split(",")    # ['a','b','c']
"hi there".split()    # ['hi','there']
",".join(['a','b'])   # 'a,b'
"".join(['h','i'])    # 'hi'

Replace & Format

s.replace("l","r")     # all occurrences
s.replace("l","r",1)   # first only
f"val={x}"             # f-string (fastest)

Char / Ord

ord('a')               # 97
chr(97)                # 'a'
ord(c) - ord('a')      # 0-25 index

Checks

s.isalpha()   # all letters
s.isdigit()   # all digits
s.isalnum()   # letters or digits
s.isspace()   # all whitespace

Loop Patterns

# forward — char by char
for ch in s: ...

# forward — with index
for i in range(len(s)): ...

# backward
for i in range(len(s)-1, -1, -1): ...

# every other char
for i in range(0, len(s), 2): ...

# index + char together
for i, ch in enumerate(s): ...

# two pointers on string — palindrome check
l, r = 0, len(s) - 1
while l < r:
    if s[l] != s[r]: ...
    l += 1; r -= 1

# sliding window (fixed size k)
for i in range(len(s) - k + 1):
    window = s[i:i+k]

Common Patterns

# Palindrome
s == s[::-1]

# Anagram check
Counter(s) == Counter(t)

# Frequency map
from collections import Counter
freq = Counter(s)

# Build string efficiently (avoid repeated +)
parts = []
parts.append(x)
result = ''.join(parts)   # O(n)
# repeated s += x is O(n^2) in a loop
