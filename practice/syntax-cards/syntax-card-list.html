LIST — Python Syntax Card

Create & Initialize

a = []                       # empty
a = [0] * n                  # n zeros
a = list(range(n))           # [0,1,...,n-1]
a = [x*2 for x in range(5)] # comprehension

Access & Slice

a[0]      # first
a[-1]     # last
a[1:4]    # index 1,2,3 (end exclusive)
a[::-1]   # reversed copy
a[::2]    # every other element

Add / Remove

a.append(x)       # end        O(1)
a.insert(i, x)    # at index   O(n)
a.pop()           # last       O(1)
a.pop(i)          # at index   O(n)
a.remove(x)       # first x    O(n)
a.extend([1,2])   # add many   O(k)

Sort & Order

a.sort()                      # in-place
a.sort(reverse=True)
a.sort(key=lambda x: x[1])   # by field
sorted(a)                     # new list
a.reverse()                   # in-place

Stack Pattern

stack = []
stack.append(x)   # push O(1)
stack.pop()       # pop  O(1)
stack[-1]         # peek O(1)

Search & Info

x in a            # membership  O(n)
a.index(x)        # first index O(n)
a.count(x)        # occurrences O(n)
len(a)            # length      O(1)
min(a), max(a)    # O(n)
sum(a)            # O(n)

Copy

b = a[:]          # shallow copy
b = a.copy()      # shallow copy
import copy
b = copy.deepcopy(a)  # deep (nested)

2D List

grid = [[0]*c for _ in range(r)]  # correct
grid = [[0]*c] * r                # WRONG — shared rows
grid[r][c]        # access cell
len(grid)         # row count
len(grid[0])      # col count

Loop Patterns

# forward — value
for x in a: ...

# forward — index
for i in range(len(a)): ...

# backward
for i in range(len(a)-1, -1, -1): ...

# every other (step)
for i in range(0, len(a), 2):  # even indices
for i in range(1, len(a), 2):  # odd indices

# index + value together
for i, v in enumerate(a): ...

# two pointers — squeeze inward
l, r = 0, len(a) - 1
while l < r:
    ...
    l += 1; r -= 1

# adjacent pairs (i, i+1)
for i in range(len(a) - 1): ...

# zip two lists
for x, y in zip(a, b): ...

# all pairs (i < j) — O(n²)
for i in range(len(a)):
    for j in range(i+1, len(a)): ...

Other Patterns

# flatten 2D
flat = [x for row in grid for x in row]

# unpack
first, *rest = a
*init, last  = a
